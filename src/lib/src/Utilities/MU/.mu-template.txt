	if ( file_exists( $shieldFile ) ) {
		require_once( $shieldFile );

		add_action( 'admin_notices', function () {

				'<div id="message" class="error">'.
				'<p>The Shield plugin appears to have been removed. Please also remove the MU-Plugin file: <code>%s</code>.</p></div>',
				esc_html( __FILE__ )
			);
		} );
	}
} );
